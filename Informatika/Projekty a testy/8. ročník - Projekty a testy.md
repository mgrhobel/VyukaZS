```python
def statistika(body):
    celkem = 0
    for b in body.values():
        celkem += b
    prumer = celkem / len(body)
    uspesni = 0
    for b in body.values():
        if b >= 75:
            uspesni += 1
    return prumer, uspesni
```

### Příklad struktury (varianta B – Inventář)

```python
inventar = []

def pridej_polozku(nazev, pocet, cena):
    inventar.append({"nazev": nazev, "pocet": pocet, "cena": cena})

def zobraz_vse():
    if not inventar:
        print("Inventář je prázdný.")
        return
    for p in inventar:
        print(f"{p['nazev']}: {p['pocet']} ks @ {p['cena']} Kč")

def hledej(nazev):
    vysledky = [p for p in inventar if nazev.lower() in p["nazev"].lower()]
    return vysledky

def celkova_hodnota():
    return sum(p["pocet"] * p["cena"] for p in inventar)

# Menu
while True:
    print("\n1) Přidat  2) Zobrazit  3) Hledat  4) Celková hodnota  0) Konec")
    volba = input("Volba: ")
    if volba == "1":
        n = input("Název: ")
        pocet_str = input("Počet: ")
        cena_str = input("Cena: ")
        if not n:
            print("Název nesmí být prázdný.")
        elif not pocet_str.isdigit():
            print("Počet musí být celé číslo.")
        elif not cena_str.replace(".", "", 1).isdigit():
            print("Cena musí být číslo.")
        else:
            pridej_polozku(n, int(pocet_str), float(cena_str))
    elif volba == "2":
        zobraz_vse()
    elif volba == "3":
        q = input("Hledej: ")
        for p in hledej(q):
            print(p)
    elif volba == "4":
        print(f"Celková hodnota: {celkova_hodnota():.2f} Kč")
    elif volba == "0":
        break
```
